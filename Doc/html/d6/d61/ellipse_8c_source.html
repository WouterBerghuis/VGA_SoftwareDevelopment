165 ELLIPSE_ERROR_CODES API_Calculate_Point_On_Ellipse(uint16_t x_coor_center, uint16_t y_coor_center, float width, float height,
166  uint16_t x_coor_ellipse, uint16_t *y_coor_ellipse){
167 
168  float radius_width_sqrt = __Square_Number((float)(width / 2));
169  float radius_height_sqrt = __Square_Number((float)(height / 2));
170 
171  float formula_section_one = radius_width_sqrt * radius_height_sqrt;
172 
173  float formula_section_two = x_coor_ellipse - x_coor_center;
174  formula_section_two = __Square_Number(formula_section_two);
175  formula_section_two *= radius_height_sqrt;
176 
177  float formula_section_three = formula_section_one - formula_section_two;
178  formula_section_three /= radius_width_sqrt;
179 
180  if(formula_section_three < 0)
181  *y_coor_ellipse = y_coor_center;
182  else{
183  formula_section_three = sqrt(formula_section_three);
184  *y_coor_ellipse = (round(formula_section_three) + y_coor_center);
185  }
186 
187  return CALCULATE_POINT_ON_ELLIPSE_SUCCESS;
